async function loadPrices() { const url = 'https://api.coingecko.com/api/v3/simple/price?ids=bitcoin,ethereum,cardano&vs_currencies=usd&include_24hr_change=true'; const res = await fetch(url); const data = await res.json(); document.getElementById('prices').innerHTML = `

BTC: $${data.bitcoin.usd} (${data.bitcoin.usd_24h_change.toFixed(2)}%)

ETH: $${data.ethereum.usd} (${data.ethereum.usd_24h_change.toFixed(2)}%)

ADA: $${data.cardano.usd} (${data.cardano.usd_24h_change.toFixed(2)}%)

`; } loadPrices(); setInterval(loadPrices, 20000); function showChart() { document.getElementById('chartContainer').style.display = 'block'; new TradingView.widget({ width: '100%', height: 500, symbol: 'BINANCE:BTCUSDT', interval: '60', theme: 'dark', style: '1', locale: 'es', container_id: 'chart', }); } function showSignals() { const closes = [68000, 68200, 67800, 67900, 68100, 68300]; const rsi = calcRSI(closes); const ema20 = calcEMA(closes, 20); const ema50 = calcEMA(closes, 50); let signal = 'Neutral'; if (rsi < 30) signal = 'Compra (RSI bajo)'; if (rsi > 70) signal = 'Venta (RSI alto)'; if (ema20 > ema50) signal = 'Tendencia Alcista'; if (ema20 < ema50) signal = 'Tendencia Bajista'; document.getElementById('signals').style.display = 'block'; document.getElementById('signals').innerHTML = `

Panel de Señales

RSI: ${rsi.toFixed(2)}

EMA20: ${ema20.toFixed(2)}

EMA50: ${ema50.toFixed(2)}

Señal: ${signal}

`; } function calcEMA(values, period) { let k = 2 / (period + 1); let ema = values[0]; for (let i = 1; i < values.length; i++) { ema = values[i] * k + ema * (1 - k); } return ema; } function calcRSI(values) { let gains = 0, losses = 0; for (let i = 1; i < values.length; i++) { const change = values[i] - values[i - 1]; if (change > 0) gains += change; else losses -= change; } const rs = gains / (losses || 1); return 100 - 100 / (1 + rs); }
